/*

Ammo - Buckshot Shells

The ammo used for shotguns. Does a spread of 5 projectiles.

*/

// Ammo Class
/*
Base class for pickups of this ammo type.
This is also used to store the amount of the ammo overall.
*/
actor shell_buckshot : Ammo {
	Tag "$AT_SHELLBUCKSHOT"
	
	Inventory.Icon "SHELA0"
	
	Inventory.MaxAmount 100
}

// Ammo Pickups
actor shell_buckshot_some: shell_buckshot {
	Inventory.PickupMessage "$APU_SHELLBUCKSHOT_SOME"
	
	Inventory.Amount 4
	
	States {
		Spawn:
			SHEL A -1
			Stop
		Use:
			TNT1 A 0
	}
}

// Projectile
actor shell_buckshot_p : FastProjectile {
	Tag "shell_buckshot_p"
	
	Height 2
	Radius 1
	Speed 25
	Damage (Random(2, 4) * 5)
	
	
	PROJECTILE
	+BLOODSPLATTER
	+FORCEXYBILLBOARD
	-NOBLOCKMAP
	
	Decal BulletChip
		
	//DeathSound "ricochet"
	
	States {
		Spawn:
			TNT1 AA 0 A_ScaleVelocity (1.0 / (25*1.0))
			TNT1 A 0 A_ScaleVelocity (150 * 1.0)
		Idle:
			TSHT A 1 Bright
			Loop			
		Death: // Default
			TNT1 A 0 Bright A_SpawnItemEx("HitPuff")
			//Play a ricochet sound, alert nearby monsters.
			Stop
	}
}
